        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-card-match-app';

        let db, auth, userId;


        async function setupFirebase() {

            try {

                const firebaseConfig = JSON.parse(typeof __firebase_config !== 'undefined' ? __firebase_config : '{}');

                const app = initializeApp(firebaseConfig);

                db = getFirestore(app);

                auth = getAuth(app);

                setLogLevel('debug');

                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

                    await signInWithCustomToken(auth, __initial_auth_token);

                } else {

                    await signInAnonymously(auth);

                }

                userId = auth.currentUser?.uid || `anon-${crypto.randomUUID()}`;

                document.getElementById('userId').textContent = userId;

                listenForLeaderboardUpdates();

            } catch (error) {

                console.error("Firebase initialization failed:", error);

                document.getElementById('leaderboard').innerHTML = '<p class="text-red-400">Could not connect to leaderboard.</p>';

            }

        }


        const gameBoard = document.getElementById('game-board');

        const timerEl = document.getElementById('timer');

        const matchesEl = document.getElementById('matches');

        const modal = document.getElementById('end-game-modal');

        const finalTimeEl = document.getElementById('final-time');


        let cards = [];

        let flippedCards = [];

        let matchedPairs = 0;

        let lockBoard = false;

        let timerInterval;

        let startTime;

        

        const CARD_CONTENT = [

            { text: 'Gamer', icon: '🎮' },

            { text: 'Sports Fan', icon: '⚽️' },

            { text: 'Business Traveller', icon: '✈️' },

        ];


        const matchSound = new Tone.Synth({ oscillator: { type: 'square' }, envelope: { attack: 0.01, decay: 0.1, sustain: 0.2, release: 0.1 } }).toDestination();

        const mismatchSound = new Tone.MonoSynth({ oscillator: { type: 'sawtooth' }, envelope: { attack: 0.01, decay: 0.2, sustain: 0.1, release: 0.2 }, filterEnvelope: { attack: 0.01, decay: 0.1, sustain: 0.05, release: 0.1, baseFrequency: 200, octaves: 4 } }).toDestination();


        function playMatchSound() { Tone.start(); matchSound.triggerAttackRelease('C5', '8n'); setTimeout(() => matchSound.triggerAttackRelease('G5', '8n'), 120); }

        function playMismatchSound() { Tone.start(); mismatchSound.triggerAttackRelease('G#2', '8n'); setTimeout(() => mismatchSound.triggerAttackRelease('C2', '8n'), 120); }


        function shuffle(array) {

            for (let i = array.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [array[i], array[j]] = [array[j], array[i]];

            }

            return array;

        }


        function createGameBoard() {

            gameBoard.innerHTML = '';

            matchedPairs = 0;

            matchesEl.textContent = `0 / ${CARD_CONTENT.length}`;

            

            const cardValues = shuffle([...CARD_CONTENT, ...CARD_CONTENT]);

            

            cardValues.forEach(value => {

                const cardElement = document.createElement('div');

                cardElement.classList.add('card');

                cardElement.dataset.value = value.text;


                cardElement.innerHTML = `

                    <div class="card-face card-front">?</div>

                    <div class="card-face card-back">

                        <div class="card-icon">${value.icon}</div>

                        <div class="card-text">${value.text}</div>

                    </div>

                `;


                cardElement.addEventListener('click', handleCardClick);

                gameBoard.appendChild(cardElement);

                cards.push(cardElement);

            });

        }


        function handleCardClick(e) {

            if (!startTime) startTimer();

            const clickedCard = e.currentTarget;

            if (lockBoard || clickedCard.classList.contains('flipped') || clickedCard.classList.contains('matched')) return;

            flipCard(clickedCard);

            if (flippedCards.length === 2) checkForMatch();

        }


        function flipCard(card) {

            card.classList.add('flipped');

            flippedCards.push(card);

        }


        function checkForMatch() {

            lockBoard = true;

            const [cardOne, cardTwo] = flippedCards;

            if (cardOne.dataset.value === cardTwo.dataset.value) {

                playMatchSound();

                setTimeout(() => { disableCards(); checkWinCondition(); }, 500);

            } else {

                playMismatchSound();

                setTimeout(unflipCards, 1000);

            }

        }


        function disableCards() {

            flippedCards.forEach(card => {

                card.classList.remove('flipped');

                card.classList.add('matched');

            });

            matchedPairs++;

            matchesEl.textContent = `${matchedPairs} / ${CARD_CONTENT.length}`;

            resetBoardState();

        }


        function unflipCards() {

            flippedCards.forEach(card => card.classList.remove('flipped'));

            resetBoardState();

        }


        function resetBoardState() {

            flippedCards = [];

            lockBoard = false;

        }


        function checkWinCondition() {

            if (matchedPairs === CARD_CONTENT.length) {

                stopTimer();

                const timeTaken = ((Date.now() - startTime) / 1000).toFixed(2);

                finalTimeEl.textContent = timeTaken;

                setTimeout(() => modal.classList.remove('hidden'), 500);

            }

        }


        function startTimer() {

            startTime = Date.now();

            timerInterval = setInterval(() => {

                timerEl.textContent = ((Date.now() - startTime) / 1000).toFixed(1);

            }, 100);

        }


        function stopTimer() {

            clearInterval(timerInterval);

        }


        function resetGame() {

            stopTimer();

            startTime = null;

            timerEl.textContent = '0.0';

            modal.classList.add('hidden');

            document.getElementById('player-name').value = '';

            document.getElementById('player-company').value = '';

            document.getElementById('submit-score').disabled = false;

            document.getElementById('submit-score').textContent = 'Submit Score';

            resetBoardState();

            createGameBoard();

        }


        async function addScoreToLeaderboard(name, company, time) {

            if (!db) return console.error("Firestore is not initialized.");

            try {

                const leaderboardCollection = collection(db, `artifacts/${appId}/public/data/leaderboard`);

                await addDoc(leaderboardCollection, { name, company, time: parseFloat(time), createdAt: new Date(), userId });

            } catch (error) { console.error("Error adding document: ", error); }

        }


        function listenForLeaderboardUpdates() {

            if (!db) return console.error("Firestore is not initialized.");

            const leaderboardCollection = collection(db, `artifacts/${appId}/public/data/leaderboard`);

            onSnapshot(leaderboardCollection, (snapshot) => {

                // --- FIX: Correctly call .data() and filter out entries without a valid time ---

                const scores = snapshot.docs

                    .map(doc => doc.data())

                    .filter(score => typeof score.time === 'number')

                    .sort((a, b) => a.time - b.time)

                    .slice(0, 10);

            });

        function renderLeaderboard(scores) {

            const leaderboardEl = document.getElementById('leaderboard');

            leaderboardEl.innerHTML = '';

            if (scores.length === 0) {

                leaderboardEl.innerHTML = '<p class="text-center text-gray-400">Be the first!</p>';

                return;

            }

            scores.forEach((score, index) => {

                const entry = document.createElement('div');

                entry.className = 'leaderboard-entry flex justify-between p-2 rounded-md items-center';

                let medal = ['🥇', '🥈', '🥉'][index] || '';

                const companyHTML = score.company ? `<span class="leaderboard-company">${score.company}</span>` : '';


                entry.innerHTML = `

                    <div>

                        <span>${index + 1}. ${medal} ${score.name}</span>

                        ${companyHTML}

                    </div>

                    <span class="font-bold text-lg">${score.time.toFixed(2)}s</span>

                `;

                leaderboardEl.appendChild(entry);

            });

        }
